Class LinkedList<T>

A singlely-linked list.

const list = new LinkedList<number>()
list.push(1)
list.append([2, 3])
list
  .filter((num) => n % 2 !== 0)
  .map((num) => n * 2)
  .forEach((num) => console.log(num))
